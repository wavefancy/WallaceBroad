\newcommand\textline[4][t]{%
  \par\smallskip\noindent\parbox[#1]{.100\textwidth}{\raggedright\texttt{}#2}%
  %control the text location for the title of subfigures.
  \parbox[#1]{.302\textwidth}{\centering#3}%
  \parbox[#1]{.422\textwidth}{\raggedleft\texttt{#4}}\par\smallskip%
}
