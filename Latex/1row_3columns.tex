% https://tex.stackexchange.com/questions/55472/how-to-make-text-aligned-left-center-right-in-the-same-line/55476
\newcommand\textline[4][t]{%
  \par\smallskip\noindent\parbox[#1]{.0200\textwidth}{\raggedright\texttt{}#2}%
  %control the text location for the title of subfigures.
  %*** Only change this paramter to centering the subfigure title.
  %*** The other no effects on this.
  \parbox[#1]{.27\textwidth}{\centering#3}%
  \parbox[#1]{.022\textwidth}{\raggedleft\texttt{#4}}\par\smallskip%
}
